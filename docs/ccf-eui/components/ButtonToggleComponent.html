import {
  ChangeDetectionStrategy,
  Component,
  EventEmitter,
  Input,
  Output
} from '@angular/core';

@Component({
  selector: 'ccf-button-toggle',
  templateUrl: './button-toggle.component.html',
  styleUrls: ['./button-toggle.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush,
})
export class ButtonToggleComponent {
  @Input() menuOptions: string[];

  @Input() selectedItems?: string[] = [];

  @Input() enableTooltip = false;
  @Input() tooltips = [];

  /**
   * Any time a button is clicked, event is emitted.
   */
  @Output() readonly selectionChange = new EventEmitter<string[]>();

  isItemSelected(item: string) {
    return this.selectedItems?.includes(item);
  }

  toggleSelection(value: string): void {
    if (this.isItemSelected(value)) {
      this.selectedItems = this.selectedItems?.filter(
        (el) => el != value
      );
    } else {
      this.selectedItems?.push(value);
    }
    this.selectionChange.emit(this.selectedItems);
  }
}

<mat-button-toggle-group class="toggleGroup" name="menuOptions" aria-label="menuOptions" multiple>
  <mat-button-toggle
  [class.selected]="isItemSelected(item)"
  *ngFor="let item of menuOptions; let i=index;"
  [value]="item"
  (change)="toggleSelection($event.value)"
  >
    <div class="tooltip-content-container" [matTooltip]="enableTooltip && tooltips[i]">

      <mat-icon [class.hidden]="!isItemSelected(item)">done</mat-icon>
      <span> {{item}}</span>
    </div>
  </mat-button-toggle>
</mat-button-toggle-group>
